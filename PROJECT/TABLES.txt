CREATE TABLE BOOKING_DETAILS
(
REQUEST_ID NUMBER(10)  GENERATED AS IDENTITY,
AGENT_ID    NUMBER(10) NOT NULL,
START_DATE DATE      ,
END_DATE   DATE,
ACCOMANDATION_TYPE VARCHAR(10),
BOOKING_STATUS     VARCHAR(10),
COST_OF_ESTIMATION NUMBER(8,2),
CONSTRAINT PK_REQUEST_ID PRIMARY KEY(REQUEST_ID)
);
